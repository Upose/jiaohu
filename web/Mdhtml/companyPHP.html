    //创建数据库连接
    class DB{
        private static $_instance = NULL;
        private static $_instances = [];
        private function __construct() {
        }
        public static function getInstance ($database = DB_ID) {
            global $DATABASE_LIST;
            if (!isset(self::$_instances[$database])||is_null(self::$_instances[$database])) {
                $db = NewADOConnection('mysql');   //数据库类型
                //$db->Connect(DB_IP . ':' . DB_IP_PORT, DB_IP_USERNAME, DB_IP_PASSWORD, $database);
                $db->Connect($DATABASE_LIST[$database]['DB_IP'] . ':' . $DATABASE_LIST[$database]['DB_IP_PORT'], $DATABASE_LIST[$database]['DB_IP_USERNAME'], $DATABASE_LIST[$database]['DB_IP_PASSWORD'], $DATABASE_LIST[$database]['DB_NAME']);
                $db->SetFetchMode(ADODB_FETCH_ASSOC);
                self::$_instances[$database] = $db;
            }
            return self::$_instances[$database];
        }   
        public function __clone(){}
    }
